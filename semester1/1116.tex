\centerline {Задание №1}
\startformula \startalign[align=left]
\NC \int\frac{dx}{x-3} = \dmat{t=x-3;x=t+3} = \int\frac{dt}t = \ln{|t|} + C = \ln{|x-3|} + C \NR
\NC \int\frac{dx}{5-x} = \dmat{t=5-x;x=5-t} = -\int\frac{dt}t = -\ln{|t|} + C = -\ln{|5-x|} + C \NR
\NC \int\frac{dx}{2x-1} = \dmat{t=2x-1;x=\frac{t+1}2}
    = \frac12\int\frac{dt}t = \frac12\ln{|t|} + C = \frac12\ln{|2x-1|} + C \NR
\NC \int\frac{dx}{15-3x} = \dmat{t=15-3x;x=\frac{15-t}3}
    = -\frac13\int\frac{dt}t = -\frac13\ln{|t|} + C = -\frac13\ln{|15-3x|} + C \NR
\NC \int\frac{dx}{4-7x} = \dmat{t=4-7x;x=\frac{4-t}7}
    = -\frac17\int\frac{dt}t = -\frac17\ln{|t|} + C = -\frac17\ln{|4-7x|} + C \NR
\stopalign \stopformula

\centerline {Задание №2}
\startformula \startalign[align=left]
\NC \int\frac{dx}{x^2+4x+14} = \int\frac{dx}{(x+2)^2 + 10}
    = \dmat{t=x+2;x=t-2} = \int\frac{dt}{t^2 + 10} = \NR
\NC = \frac1{\sqrt{10}} \atan{\frac{t}{\sqrt{10}}} + C
    = \frac1{\sqrt{10}} \atan{\frac{x+2}{\sqrt{10}}} + C \NR
\NC \int\frac{dx}{x^2+8x+15} = \int\frac{dx}{(x+4)^2 - 1}
    = \dmat{t=x+4;x=t-4} = -\int\frac{dt}{1 - t^2} = \NR
\NC = -\frac12 \ln{\left|\frac{1+t}{1-t}\right|} + C
    = -\frac12 \ln{\left|\frac{5+x}{-3-x}\right|} + C \NR
\NC \int\frac{dx}{x^2+x+1} = \int\frac{dx}{\left(x+\frac12\right)^2 + \frac34}
    = \dmat{t=x+\frac12;x=t-\frac12} = \int\frac{dt}{t^2 + \frac34} = \NR
\NC = \frac2{\sqrt{3}} \atan{\frac{2t}{\sqrt{3}}} + C
    = \frac2{\sqrt{3}} \atan{\frac{2x + 1}{\sqrt{3}}} + C \NR
\NC \int\frac{dx}{x^2+12x+6} = \int\frac{dx}{(x+6)^2 - 30}
    = \dmat{t=x+6;x=t-6} = -\int\frac{dt}{30 - t^2} = \NR
\NC = -\frac1{2\sqrt{30}} \ln{\left|\frac{\sqrt{30}+t}{\sqrt{30}-t}\right|} + C
    = -\frac1{2\sqrt{30}} \ln{\left|\frac{\sqrt{30}+x+6}{\sqrt{30}-x-6}\right|} + C \NR
\stopalign \stopformula
\startformula \startalign[align=left]
\NC \int\frac{dx}{x^2-8x+14} = \int\frac{dx}{(x-4)^2-2}
    = \dmat{t=x-4;x=t+4} = -\int\frac{dt}{2 - t^2} = \NR
\NC = -\frac1{2\sqrt{2}} \ln{\left|\frac{\sqrt{2}+t}{\sqrt{2}-t}\right|} + C
    = -\frac1{2\sqrt{2}} \ln{\left|\frac{\sqrt{2}+x-4}{\sqrt{2}-x+4}\right|} + C \NR
\stopalign \stopformula

\centerline {Задание №4}
\startformula \startalign[align=left]
\NC \int\frac{dx}{x(x+7)} = \int\frac{dx}{7x} - \int\frac{dx}{7(x+7)}
    = \frac17\ln{|x|} - \frac17\ln{|x+7|} + C
    = \frac17\ln{\left|\frac{x}{x+7}\right|} + C \NR
\NC \int\frac{dx}{(8-x)(4+x)} = -\frac1{12}\int\frac{dx}{x-8} + \frac1{12}\int\frac{dx}{4+x}
    = -\frac1{12}\ln{|x-8|} + \frac1{12}\ln{|x+4|} + C \NR
\NC = \frac1{12}\ln{\left|\frac{x+4}{x-8}\right|} + C \NR
\NC \int\frac{dx}{x^2 - 81} = \int\frac{dx}{(x-9)(x+9)}
    = \frac1{18}\int\frac{dx}{x-9} - \frac1{18}\int\frac{dx}{x+9} = \NR
\NC = \frac1{18}\ln{|x-9|} - \frac1{18}\ln{|x+9|} + C
    = \frac1{18}\ln{\left|\frac{x-9}{x+9}\right|} + C \NR
\NC \int\frac{dx}{(x+7)(x-4)} = \frac1{11}\int\frac{dx}{x-4} - \frac1{11}\int\frac{dx}{x+7}
    = \frac1{11}\ln{|x-4|} - \frac1{11}\ln{|x+7|} + C \NR
\NC = \frac1{11}\ln{\left|\frac{x-4}{x+7}\right|} + C \NR
\NC \int\frac{dx}{(2+5x)(3-7x)} = \frac1{29}\int\frac{5dx}{5x+2} - \frac1{29}\int\frac{7dx}{7x-3}
    = \frac1{29}\ln{|5x+2|} - \frac1{29}\ln{|7x-3|} + C \NR
\NC = \frac1{29}\ln{\left|\frac{5x+2}{7x-3}\right|} + C \NR
\stopalign \stopformula
